        <div class="mt-12 flex-grow">
            <h1 class="text-2xl font-bold text-gray-800 mb-2">Logística de Carga</h1>
            <p class="text-sm text-gray-500 mb-8">Los sacos se cargan automáticamente de adelante hacia atrás.</p>

            <!-- Formulario de Entrada -->
            <div class="space-y-6 mb-8">
                <div>
                    <label class="block text-xs font-bold text-gray-400 uppercase tracking-widest mb-2">Tipo de Masa</label>
                    <select id="sackType" class="w-full p-3 bg-gray-50 border border-gray-200 rounded-xl focus:ring-2 focus:ring-blue-500 outline-none transition-all">
                        <option value="dura" data-color="#8B4513">Masa Dura</option>
                        <option value="blanda" data-color="#F5F5DC">Masa Blanda</option>
                        <option value="integral" data-color="#D2B48C">Masa Integral</option>
                        <option value="especial" data-color="#DEB887">Masa Especial</option>
                    </select>
                </div>
                <div>
                    <label class="block text-xs font-bold text-gray-400 uppercase tracking-widest mb-2">Cantidad de sacos a sumar</label>
                    <input type="number" id="sackQuantity" value="14" min="1" max="700" class="w-full p-3 bg-gray-50 border border-gray-200 rounded-xl focus:ring-2 focus:ring-blue-500 outline-none transition-all">
                </div>
                <button onclick="handleAutoLoad()" class="w-full bg-blue-600 hover:bg-blue-700 text-white font-bold py-4 rounded-xl shadow-md transition-all active:scale-95">
                    Sumar Sacos a la Carga
                </button>
            </div>

            <!-- Estado del Camión -->
            <div>
                <h3 class="text-xs font-bold text-gray-400 uppercase tracking-widest mb-4">Estado de los Palets (Frontal → Trasero)</h3>
                <div id="truckStatus" class="grid grid-cols-2 gap-3"></div>
                <p id="emptyTruckMsg" class="text-sm text-gray-400 italic text-center py-4 bg-gray-50 rounded-xl border border-dashed border-gray-200">Sin carga actual</p>
            </div>
        </div>

        <div class="mt-8 pt-6 border-t border-gray-100">
            <div class="flex justify-between text-sm font-black text-gray-700">
                <span>TOTAL PALETS UTILIZADOS:</span>
                <span id="totalPalletsCounter">0 / 10</span>
            </div>

    <div id="threejs-container" class="absolute inset-0"></div>

    <!-- Notification system -->
    <div id="notification" class="fixed bottom-6 right-6 px-6 py-4 rounded-2xl text-white font-bold transform translate-y-32 transition-transform duration-300 z-50 shadow-2xl"></div>

    <script>
        // --- CONSTANTES DE DISEÑO ---
        const SACOS_POR_FILA = 7;
        const TOTAL_MAX_PALLET = 70;
        const MAX_PALLETS_PER_TRUCK = 10;

        const W = 1.333; // Ancho del saco
        const L = 2.0;   // Largo del saco
        const H = 0.42;  // Altura del saco
        
        const TOTAL_WIDTH = L + (2 * W); 
        const TOTAL_DEPTH = 3 * W;       

        // --- ESTADO ---
        let loadedPallets = []; // Array de objetos { sacks: N, color: Hex, type: Str, mesh: THREE.Group }
        let isMenuOpen = true;

        // --- THREE.JS SETUP ---
        let scene, camera, renderer, truckContainer;

        function init3D() {
            const container = document.getElementById('threejs-container');
            scene = new THREE.Scene();
            scene.background = new THREE.Color(0xf1f5f9);

            camera = new THREE.PerspectiveCamera(40, window.innerWidth / window.innerHeight, 0.1, 1000);
            camera.position.set(25, 20, 25);
            camera.lookAt(0, 2, -2);

            renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.shadowMap.enabled = true;
            renderer.shadowMap.type = THREE.PCFSoftShadowMap;
            container.appendChild(renderer.domElement);

            const ambientLight = new THREE.AmbientLight(0xffffff, 0.7);
            scene.add(ambientLight);

            const directionalLight = new THREE.DirectionalLight(0xffffff, 1.0);
            directionalLight.position.set(30, 50, 30);
            directionalLight.castShadow = true;
            directionalLight.shadow.mapSize.width = 2048;
            directionalLight.shadow.mapSize.height = 2048;
            scene.add(directionalLight);

            const grid = new THREE.GridHelper(100, 50, 0xcbd5e1, 0xe2e8f0);
            scene.add(grid);

            createTruck();
            setupMenu();

            window.addEventListener('resize', onWindowResize, false);
            animate();
        }

        function setupMenu() {
            const btn = document.getElementById('menuToggle');
            const sidebar = document.getElementById('sidebar');
            const icon = document.getElementById('menuIcon');

            btn.addEventListener('click', () => {
                isMenuOpen = !isMenuOpen;
                if(isMenuOpen) {
                    sidebar.classList.remove('menu-hidden');
                    icon.innerHTML = '<path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12" />';
                } else {
                    sidebar.classList.add('menu-hidden');
                    icon.innerHTML = '<path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M4 6h16M4 12h16M4 18h16" />';
                }
            });
        }

        function createCustomPallet() {
            const woodMat = new THREE.MeshPhongMaterial({ color: 0x4a3728 });
            const palletBase = new THREE.Group();
            const slatWidth = TOTAL_WIDTH + 0.1;
            const slatDepth = 0.5;
            
            for(let i = 0; i < 6; i++) {
                const slat = new THREE.Mesh(new THREE.BoxGeometry(slatWidth, 0.15, slatDepth), woodMat);
                slat.position.set(0, 0, (i * (TOTAL_DEPTH/5)) - (TOTAL_DEPTH/2));
                slat.castShadow = true;
                slat.receiveShadow = true;
                palletBase.add(slat);
            }

            const beamGeom = new THREE.BoxGeometry(slatWidth, 0.3, 0.2);
            for(let i = -1; i <= 1; i++) {
                const beam = new THREE.Mesh(beamGeom, woodMat);
                beam.position.set(0, -0.22, i * (TOTAL_DEPTH/2 - 0.15));
                palletBase.add(beam);
            }
            return palletBase;
        }

        function createBag(typeColor) {
            const geometry = new THREE.BoxGeometry(1, 1, 1);
            const material = new THREE.MeshPhongMaterial({ color: 0xffffff, shininess: 30 });
            const mesh = new THREE.Mesh(geometry, material);
            mesh.castShadow = true;
            mesh.receiveShadow = true;
            
            const edges = new THREE.EdgesGeometry(geometry);
            const line = new THREE.LineSegments(edges, new THREE.LineBasicMaterial({ color: 0x333333 }));
            mesh.add(line);

            const tag = new THREE.Mesh(
                new THREE.PlaneGeometry(0.5, 0.4),
                new THREE.MeshBasicMaterial({ color: typeColor, side: THREE.DoubleSide })
            );
            tag.position.y = 0.51;
            tag.rotation.x = -Math.PI / 2;
            mesh.add(tag);
            return mesh;
        }

        function createTruck() {
            truckContainer = new THREE.Group();
            const baseWidth = 12;
            const baseLength = 26;

            // Chasis
            const base = new THREE.Mesh(
                new THREE.BoxGeometry(baseWidth, 0.8, baseLength), 
                new THREE.MeshPhongMaterial({ color: 0x0f172a })
            );
            base.position.y = 0.4;
            truckContainer.add(base);

            // Cabina (Más adelante y compacta)
            const cab = new THREE.Mesh(
                new THREE.BoxGeometry(11, 7.5, 5),
                new THREE.MeshPhongMaterial({ color: 0x1e40af })
            );
            cab.position.set(0, 4.1, 10.5); 
            truckContainer.add(cab);

            // Área de carga
            const cargo = new THREE.Mesh(
                new THREE.BoxGeometry(10.5, 8, 17),
                new THREE.MeshPhongMaterial({ color: 0x64748b, wireframe: true, transparent: true, opacity: 0.1 })
            );
            cargo.position.set(0, 4.4, -2.5);
            truckContainer.add(cargo);

            scene.add(truckContainer);
        }

        // --- LÓGICA DE CARGA AUTOMÁTICA ---

        function handleAutoLoad() {
            const qtyInput = document.getElementById('sackQuantity');
            let qtyToAdd = parseInt(qtyInput.value);
            
            if (isNaN(qtyToAdd) || qtyToAdd <= 0) return;

            const select = document.getElementById('sackType');
            const color = select.options[select.selectedIndex].dataset.color;
            const typeText = select.options[select.selectedIndex].text;

            while (qtyToAdd > 0) {
                // Buscar el palet que no esté lleno
                let targetPallet = loadedPallets.find(p => p.sacks < TOTAL_MAX_PALLET);

                if (!targetPallet) {
                    if (loadedPallets.length >= MAX_PALLETS_PER_TRUCK) {
                        showNotification('¡Camión lleno! No caben más palets.', 'error');
                        break;
                    }
                    // Crear nuevo palet
                    targetPallet = {
                        sacks: 0,
                        color: color,
                        type: typeText,
                        mesh: new THREE.Group()
                    };
                    loadedPallets.push(targetPallet);
                    initPalletIn3D(targetPallet, loadedPallets.length - 1);
                }

                let spaceAvailable = TOTAL_MAX_PALLET - targetPallet.sacks;
                let amountToFill = Math.min(qtyToAdd, spaceAvailable);

                targetPallet.sacks += amountToFill;
                qtyToAdd -= amountToFill;

                // Actualizar visualmente el palet específico
                updatePalletSacks3D(targetPallet);
            }

            updateUI();
            showNotification('Carga actualizada');
        }

        function initPalletIn3D(palletObj, index) {
            const group = palletObj.mesh;
            
            // Posicionamiento 2x5 de ADELANTE hacia ATRÁS
            // Frontal es Z positivo (cerca de cabina), Trasero es Z negativo
            const tRow = Math.floor(index / 2); // 0 a 4
            const tCol = index % 2; // 0 o 1
            
            const pX = tCol === 0 ? -2.6 : 2.6;
            // Adelante (Z=5) hacia atrás (Z=-11)
            const pZ = 5 - (tRow * 4.2);

            group.position.set(pX, 1.2, pZ);
            group.scale.set(0.9, 0.9, 0.9);
            
            truckContainer.add(group);
        }

        function updatePalletSacks3D(palletObj) {
            const group = palletObj.mesh;
            group.clear(); // Limpiar sacos anteriores

            // Añadir base de madera
            group.add(createCustomPallet());

            // Añadir sacos actuales
            for (let i = 0; i < palletObj.sacks; i++) {
                const layerIndex = Math.floor(i / SACOS_POR_FILA);
                const posInLayer = i % SACOS_POR_FILA;
                const isEvenLayer = layerIndex % 2 === 0;

                const bag = createBag(palletObj.color);
                
                if (posInLayer < 4) {
                    const row = Math.floor(posInLayer / 2);
                    const col = posInLayer % 2;
                    bag.scale.set(W, H, L);
                    let x = (TOTAL_WIDTH / 2) - L - (W / 2) - (col * W);
                    let z = (row === 0 ? -L/2 : L/2);
                    if (!isEvenLayer) x = -x;
                    bag.position.set(x, (layerIndex * (H + 0.02)) + 0.15, z);
                } 
                else {
                    const hIndex = posInLayer - 4;
                    bag.scale.set(L, H, W);
                    let x = (TOTAL_WIDTH / 2) - (L / 2);
                    let z = (hIndex - 1) * W;
                    if (!isEvenLayer) x = -x;
                    bag.position.set(x, (layerIndex * (H + 0.02)) + 0.15, z);
                }
                group.add(bag);
            }
        }

        function updateUI() {
            document.getElementById('totalPalletsCounter').innerText = `${loadedPallets.length} / ${MAX_PALLETS_PER_TRUCK}`;
            
            const truckStatus = document.getElementById('truckStatus');
            const emptyMsg = document.getElementById('emptyTruckMsg');
            
            if (loadedPallets.length > 0) {
                emptyMsg.style.display = 'none';
                truckStatus.innerHTML = loadedPallets.map((p, i) => `
                    <div class="flex flex-col items-center ${p.sacks === 70 ? 'bg-green-50 border-green-200' : 'bg-blue-50 border-blue-200'} p-3 rounded-xl border shadow-sm transition-all">
                        <div class="w-full flex justify-between mb-2">
                            <span class="font-black text-[10px] ${p.sacks === 70 ? 'text-green-400' : 'text-blue-400'}">P${i + 1}</span>
                            <div class="w-3 h-3 rounded-full shadow-inner" style="background-color: ${p.color}"></div>
                        </div>
                        <div class="w-full bg-gray-200 rounded-full h-1.5 mb-1">
                            <div class="bg-blue-600 h-1.5 rounded-full" style="width: ${(p.sacks/70)*100}%"></div>
                        </div>
                        <span class="text-[10px] font-bold text-gray-700">${p.sacks} / 70</span>
                    </div>
                `).join('');
            }
        }

        function showNotification(msg, type = 'success') {
            const n = document.getElementById('notification');
            n.innerText = msg;
            n.style.backgroundColor = type === 'success' ? '#10b981' : '#ef4444';
            n.classList.remove('translate-y-32');
            setTimeout(() => n.classList.add('translate-y-32'), 3000);
        }

        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        function animate() {
            requestAnimationFrame(animate);
            if(truckContainer) truckContainer.rotation.y += 0.0008;
            renderer.render(scene, camera);
        }

        window.onload = init3D;
    </script>
